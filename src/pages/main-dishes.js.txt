    ingredients: [

    ],

    ],

    ],

export default function MainDishes() {
  const [openDish, setOpenDish] = useState(null);

  const toggleDish = (index) => {
    setOpenDish(openDish === index ? null : index);
  };

  return (
    <div className="p-6">
      <h1 className="text-3xl font-bold mb-4">Main Dishes</h1>
      <div className="space-y-4">
        {dishes.map((dish, index) => (
          <div key={index} className="border rounded-lg shadow-md p-4">
            <button
              onClick={() => toggleDish(index)}
              className="w-full text-left font-semibold text-lg flex justify-between items-center"
            >
              {dish.name}
              <span>{openDish === index ? "▲" : "▼"}</span>
            </button>
            {openDish === index && (
              <div className="mt-4">
                <img
                  src={dish.image}
                  alt={dish.name}
                  className="w-full rounded-lg mb-4"
                />
                <h2 className="text-xl font-semibold">Ingredients</h2>
                <ul className="list-disc list-inside mb-4">
                  {dish.ingredients.map((item, i) => (
                    <li key={i}>{item}</li>
                  ))}
                </ul>
                <h2 className="text-xl font-semibold">Cooking Steps</h2>
                <ol className="list-decimal list-inside">
                  {dish.steps.map((step, i) => (
                    <li key={i}>{step}</li>
                  ))}
                </ol>
              </div>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}
